# AI模拟面试平台 - 数据表设计

## 1. 数据库选型说明

- **关系数据库**: PostgreSQL 14+ (主数据存储)
- **向量数据库**: Chroma DB (知识库向量检索)
- **缓存数据库**: Redis (会话、缓存)

## 2. 核心数据表设计

### 2.1 用户相关表

#### 2.1.1 用户表 (users)

存储用户基本信息,支持多种登录方式扩展。

```sql
CREATE TABLE users (
    -- 基础字段
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE,
    phone VARCHAR(20) UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    
    -- 用户信息
    real_name VARCHAR(50),
    avatar_url VARCHAR(500),
    gender VARCHAR(10),
    birth_date DATE,
    school VARCHAR(100),
    major VARCHAR(100),
    graduation_year INTEGER,
    
    -- 第三方登录预留字段
    wechat_openid VARCHAR(100) UNIQUE,
    wechat_unionid VARCHAR(100),
    qq_openid VARCHAR(100) UNIQUE,
    github_id VARCHAR(100) UNIQUE,
    
    -- 账号状态
    status VARCHAR(20) DEFAULT 'active', -- active, inactive, banned
    email_verified BOOLEAN DEFAULT FALSE,
    phone_verified BOOLEAN DEFAULT FALSE,
    
    -- 角色权限
    role VARCHAR(20) DEFAULT 'user', -- user, admin, vip
    
    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_login_at TIMESTAMP,
    deleted_at TIMESTAMP, -- 软删除
    
    -- 索引
    INDEX idx_username (username),
    INDEX idx_email (email),
    INDEX idx_wechat_openid (wechat_openid),
    INDEX idx_status (status),
    INDEX idx_created_at (created_at)
);

COMMENT ON TABLE users IS '用户基础信息表';
COMMENT ON COLUMN users.wechat_openid IS '微信OpenID,预留用于微信登录';
COMMENT ON COLUMN users.github_id IS 'GitHub用户ID,预留用于GitHub登录';
```

#### 2.1.2 用户画像表 (user_profiles)

存储用户能力画像和学习偏好。

```sql
CREATE TABLE user_profiles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 目标岗位
    target_positions JSONB, -- ["Java后端", "Python后端"]
    
    -- 技能标签
    skill_tags JSONB, -- {"Java": 80, "Spring": 70, "MySQL": 75}
    
    -- 能力评估
    technical_score INTEGER DEFAULT 0, -- 技术能力分数 0-100
    expression_score INTEGER DEFAULT 0, -- 表达能力分数 0-100
    logic_score INTEGER DEFAULT 0, -- 逻辑思维分数 0-100
    comprehensive_score INTEGER DEFAULT 0, -- 综合分数 0-100
    
    -- 学习偏好
    preferred_difficulty VARCHAR(20), -- easy, medium, hard
    preferred_question_types JSONB, -- ["technical", "scenario", "algorithm"]
    
    -- 统计数据
    total_interviews INTEGER DEFAULT 0,
    total_questions_answered INTEGER DEFAULT 0,
    total_study_hours DECIMAL(10, 2) DEFAULT 0,
    
    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(user_id)
);

COMMENT ON TABLE user_profiles IS '用户画像与能力档案';
```

#### 2.1.3 用户登录日志表 (user_login_logs)

```sql
CREATE TABLE user_login_logs (

    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    login_type VARCHAR(20), -- password, wechat, github
    ip_address VARCHAR(50),
    user_agent TEXT,
    device_type VARCHAR(20), -- web, mobile, desktop
    
    login_status VARCHAR(20), -- success, failed

    INDEX idx_user_id (user_id),
    INDEX idx_created_at (created_at)

```

### 2.2 岗位相关表

#### 2.2.1 岗位表 (positions)

```sql
CREATE TABLE positions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 基本信息
    name VARCHAR(100) NOT NULL, -- Java后端工程师
    code VARCHAR(50) UNIQUE NOT NULL, -- java_backend
    category VARCHAR(50), -- 后端开发, 前端开发, 算法工程师
    
    -- 岗位描述
    description TEXT,
    requirements TEXT, -- 岗位要求
    
    -- 技能要求
    required_skills JSONB, -- ["Java", "Spring Boot", "MySQL", "Redis"]
    skill_weights JSONB, -- {"Java": 0.3, "Spring Boot": 0.25, ...}
    
    -- 难度级别
    difficulty_level VARCHAR(20), -- entry, junior, mid, senior
    
    -- 关联配置
    default_question_count INTEGER DEFAULT 10,
    default_duration INTEGER DEFAULT 30, -- 分钟
    
    -- 状态
    status VARCHAR(20) DEFAULT 'active', -- active, inactive
    
    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP,
    
    INDEX idx_code (code),
    INDEX idx_category (category),
    INDEX idx_status (status)
);

COMMENT ON TABLE positions IS '岗位信息表';
```

#### 2.2.2 岗位知识点表 (position_knowledge_points)

```sql
CREATE TABLE position_knowledge_points (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    position_id UUID NOT NULL REFERENCES positions(id) ON DELETE CASCADE,
    
    -- 知识点信息
    name VARCHAR(100) NOT NULL, -- Java集合框架
    category VARCHAR(50), -- 基础知识, 框架应用, 项目经验
    description TEXT,
    
    -- 重要程度
    importance INTEGER DEFAULT 5, -- 1-10
    
    -- 关联资源
    related_docs JSONB, -- 关联的知识库文档ID列表
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_position_id (position_id)
);
```

### 2.3 题库相关表

#### 2.3.1 题目表 (questions)

```sql
CREATE TABLE questions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 题目内容
    title VARCHAR(200) NOT NULL,
    content TEXT NOT NULL,
    
    -- 题目分类
    type VARCHAR(50) NOT NULL, -- technical(技术), scenario(场景), algorithm(算法), behavioral(行为)
    category VARCHAR(50), -- Java基础, 数据库, 网络, 算法
    
    -- 难度
    difficulty VARCHAR(20) NOT NULL, -- easy, medium, hard
    difficulty_score INTEGER, -- 1-10数值化难度
    
    -- 标签
    tags JSONB, -- ["多线程", "并发", "synchronized"]
    
    -- 参考答案
    reference_answer TEXT,
    answer_key_points JSONB, -- ["要点1", "要点2", "要点3"]
    
    -- 评分标准
    scoring_criteria JSONB, -- {"技术准确性": 40, "深度": 30, "表达": 30}
    
    -- 追问题目
    follow_up_questions JSONB, -- [question_id1, question_id2]
    
    -- 统计数据
    usage_count INTEGER DEFAULT 0,
    avg_score DECIMAL(5, 2),
    
    -- 状态
    status VARCHAR(20) DEFAULT 'active',
    
    -- 创建者
    created_by UUID REFERENCES users(id),
    
    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP,
    
    INDEX idx_type (type),
    INDEX idx_difficulty (difficulty),
    INDEX idx_category (category),
    INDEX idx_status (status)
);

COMMENT ON TABLE questions IS '面试题库';
```

#### 2.3.2 岗位题目关联表 (position_questions)

```sql
CREATE TABLE position_questions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    position_id UUID NOT NULL REFERENCES positions(id) ON DELETE CASCADE,
    question_id UUID NOT NULL REFERENCES questions(id) ON DELETE CASCADE,
    
    -- 关联权重
    weight DECIMAL(5, 2) DEFAULT 1.0, -- 该题目在该岗位的重要程度
    priority INTEGER DEFAULT 5, -- 推荐优先级 1-10
    
    -- 出现频率控制
    frequency_limit INTEGER, -- 限制出现频率
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(position_id, question_id),
    INDEX idx_position_id (position_id),
    INDEX idx_question_id (question_id)
);
```

### 2.4 面试会话相关表

#### 2.4.1 面试会话表 (interview_sessions)

```sql
CREATE TABLE interview_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 关联信息
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    position_id UUID NOT NULL REFERENCES positions(id),
    
    -- 会话配置
    session_config JSONB, -- {"difficulty": "medium", "duration": 30, "question_count": 10}
    
    -- 会话状态
    status VARCHAR(20) DEFAULT 'pending', -- pending, in_progress, completed, cancelled
    
    -- 时间信息
    scheduled_at TIMESTAMP, -- 预约时间
    started_at TIMESTAMP,
    ended_at TIMESTAMP,
    duration_seconds INTEGER, -- 实际时长(秒)
    
    -- 面试模式
    interview_mode VARCHAR(20) DEFAULT 'ai', -- ai(AI面试官), mock(模拟真人), practice(练习模式)
    
    -- AI数字人微服务会话ID
    avatar_session_id VARCHAR(100),
    
    -- 统计数据
    total_questions INTEGER DEFAULT 0,
    answered_questions INTEGER DEFAULT 0,
    
    -- 综合评分
    overall_score DECIMAL(5, 2),
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_user_id (user_id),
    INDEX idx_position_id (position_id),
    INDEX idx_status (status),
    INDEX idx_created_at (created_at)
);

COMMENT ON TABLE interview_sessions IS '面试会话记录';
```

#### 2.4.2 面试问答记录表 (interview_qa_records)

```sql
CREATE TABLE interview_qa_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 关联信息
    session_id UUID NOT NULL REFERENCES interview_sessions(id) ON DELETE CASCADE,
    question_id UUID NOT NULL REFERENCES questions(id),
    
    -- 问题序号
    question_order INTEGER, -- 第几个问题
    
    -- 问题内容快照(防止题目被修改)
    question_snapshot JSONB,
    
    -- 用户回答
    answer_text TEXT,
    answer_audio_url VARCHAR(500), -- 语音回答URL
    answer_duration_seconds INTEGER,
    
    -- 回答时间
    asked_at TIMESTAMP,
    answered_at TIMESTAMP,
    thinking_time_seconds INTEGER, -- 思考时间
    
    -- 评估结果
    score DECIMAL(5, 2),
    evaluation_result JSONB, -- 详细评估结果
    
    -- AI反馈
    ai_feedback TEXT,
    strengths JSONB, -- 优点
    weaknesses JSONB, -- 不足
    
    -- 是否追问
    is_follow_up BOOLEAN DEFAULT FALSE,
    parent_qa_id UUID REFERENCES interview_qa_records(id),
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_session_id (session_id),
    INDEX idx_question_id (question_id),
    INDEX idx_created_at (created_at)
);

COMMENT ON TABLE interview_qa_records IS '面试问答详细记录';
```

### 2.5 评估报告相关表

```sql
CREATE TABLE evaluation_reports (

    -- 多维度评分
    technical_score DECIMAL(5, 2), -- 技术能力
    expression_score DECIMAL(5, 2), -- 表达能力
    logic_score DECIMAL(5, 2), -- 逻辑思维
    depth_score DECIMAL(5, 2), -- 深度理解
    breadth_score DECIMAL(5, 2), -- 知识广度
    
    -- 能力雷达图数据
    radar_chart_data JSONB, -- {"Java基础": 85, "框架应用": 75, ...}
    
    -- 详细分析
    strengths JSONB, -- 优势分析
    weaknesses JSONB, -- 劣势分析
    improvement_suggestions JSONB, -- 改进建议
    
    -- 知识点掌握情况
    knowledge_mastery JSONB, -- {"多线程": {"score": 80, "status": "good"}, ...}
    
    -- 对比数据
    percentile_rank INTEGER, -- 百分位排名
    compared_to_average JSONB, -- 与平均水平对比
    
    -- 报告内容
    summary TEXT, -- 总结
    detailed_analysis TEXT, -- 详细分析
    
    -- 报告状态
    status VARCHAR(20) DEFAULT 'generated', -- generated, viewed, downloaded
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(session_id),
    INDEX idx_user_id (user_id),
    INDEX idx_position_id (position_id),
    INDEX idx_created_at (created_at)
);

COMMENT ON TABLE evaluation_reports IS '面试评估报告';
```

### 2.6 知识库相关表

#### 2.6.1 知识文档表 (knowledge_documents)

```sql
CREATE TABLE knowledge_documents (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 文档信息
    title VARCHAR(200) NOT NULL,
    content TEXT NOT NULL,
    summary TEXT, -- 摘要
    
    -- 文档类型
    doc_type VARCHAR(50), -- article, tutorial, reference, book_chapter
    
    -- 分类标签
    category VARCHAR(50), -- Java基础, Spring框架, 数据库
    tags JSONB, -- ["JVM", "垃圾回收", "性能优化"]
    
    -- 关联岗位
    related_positions JSONB, -- [position_id1, position_id2]
    
    -- 难度级别
    difficulty VARCHAR(20), -- beginner, intermediate, advanced
    
    -- 原始文件信息
    original_filename VARCHAR(200),
    file_url VARCHAR(500),
    file_type VARCHAR(20), -- pdf, docx, markdown, html
    
    -- 向量化状态
    vectorized BOOLEAN DEFAULT FALSE,
    vector_ids JSONB, -- Chroma中的向量ID列表
    chunk_count INTEGER DEFAULT 0, -- 分块数量
    
    -- 质量评分
    quality_score INTEGER, -- 1-10
    
    -- 统计数据
    view_count INTEGER DEFAULT 0,
    reference_count INTEGER DEFAULT 0, -- 被引用次数
    
    -- 状态
    status VARCHAR(20) DEFAULT 'active',
    
    -- 创建者
    created_by UUID REFERENCES users(id),
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP,
    
    INDEX idx_category (category),
    INDEX idx_vectorized (vectorized),
    INDEX idx_status (status)
);

COMMENT ON TABLE knowledge_documents IS '知识库文档表';
COMMENT ON COLUMN knowledge_documents.vector_ids IS 'Chroma向量数据库中的ID列表';
```

## 6. 索引优化建议

### 6.1 高频查询索引

```sql
-- 用户查询优化
CREATE INDEX idx_users_email_status ON users(email, status) WHERE deleted_at IS NULL;

-- 面试会话查询优化
CREATE INDEX idx_sessions_user_status ON interview_sessions(user_id, status, created_at DESC);

-- 问答记录查询优化
CREATE INDEX idx_qa_session_order ON interview_qa_records(session_id, question_order);

-- 题目检索优化
CREATE INDEX idx_questions_search ON questions(type, difficulty, status) 
    WHERE deleted_at IS NULL;
```

### 6.2 全文搜索索引

```sql
-- 题目内容全文搜索
CREATE INDEX idx_questions_fulltext ON questions 
    USING gin(to_tsvector('chinese', title || ' ' || content));

-- 知识文档全文搜索
CREATE INDEX idx_docs_fulltext ON knowledge_documents 
    USING gin(to_tsvector('chinese', title || ' ' || content));
```

## 8. 数据备份与归档策略

### 8.1 备份策略
- 全量备份: 每天凌晨3点
- 增量备份: 每6小时一次
- 备份保留: 30天

### 8.2 归档策略
- 6个月前的面试记录归档到冷存储
- 1年前的日志数据清理
- 软删除数据30天后物理删除
