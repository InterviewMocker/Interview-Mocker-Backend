# AI模拟面试平台 - 身份认证设计

## 1. 认证方案概述

### 1.1 认证方式
- **初期实现**: 用户名/密码登录
- **预留扩展**: 微信登录、GitHub登录、邮箱验证、手机验证码登录

### 1.2 技术选型
- **Token方案**: JWT (JSON Web Token)
- **密码加密**: bcrypt算法
- **Token存储**: Redis (黑名单机制)
- **权限模型**: RBAC (基于角色的访问控制)

## 2. JWT Token设计

### 2.1 Token结构

#### Access Token (访问令牌)
```json
{
  "header": {
    "alg": "HS256",
    "typ": "JWT"
  },
  "payload": {
    "jti": "unique_token_id",
    "sub": "user_id",
    "username": "zhangsan",
    "email": "zhangsan@example.com",
    "role": "user",
    "permissions": ["interview:create", "interview:view"],
    "iat": 1707811200,
    "exp": 1707818400,
    "type": "access"
  },
  "signature": "..."
}
```

**字段说明:**
- `jti`: Token唯一标识,用于黑名单机制
- `sub`: 用户ID
- `role`: 用户角色 (user, admin, vip)
- `permissions`: 具体权限列表
- `iat`: 签发时间
- `exp`: 过期时间 (2小时)
- `type`: Token类型

#### Refresh Token (刷新令牌)
```json
{
  "header": {
    "alg": "HS256",
    "typ": "JWT"
  },
  "payload": {
    "jti": "unique_refresh_token_id",
    "sub": "user_id",
    "iat": 1707811200,
    "exp": 1708416000,
    "type": "refresh"
  },
  "signature": "..."
}
```

**特点:**
- 有效期更长 (7天)
- 仅包含最小必要信息
- 仅用于刷新Access Token

## 3. 认证流程设计

### 3.1 用户注册流程

```
用户提交注册信息
    ↓
验证用户名/邮箱唯一性
    ↓
密码强度校验 (最少8位,包含字母数字)
    ↓
bcrypt加密密码 (cost=12)
    ↓
创建用户记录 (users表)
    ↓
创建用户画像 (user_profiles表)
    ↓
发送欢迎邮件 (异步任务)
    ↓
返回注册成功
```

### 3.3 Token刷新流程

```
客户端检测Access Token即将过期
    ↓
携带Refresh Token请求刷新
    ↓
验证Refresh Token有效性
    ↓
检查Token是否在黑名单
    ↓
验证用户状态
    ↓
生成新的Access Token
    ↓
(可选)生成新的Refresh Token
    ↓
返回新Token
```

**API接口:**
```

    "expires_in": 7200
  }
}
```

### 4.2 权限检查装饰器

```python
# core/security.py

from functools import wraps
from fastapi import HTTPException, status, Depends
from typing import List

def require_permissions(required_permissions: List[str]):
    """
    权限检查装饰器
    
    Usage:
        @router.get("/admin/users")
        @require_permissions(["user:view_all"])
        async def get_all_users():
            ...
    """
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, current_user=Depends(get_current_user), **kwargs):
            user_permissions = get_user_permissions(current_user.role)
            
            # 超级管理员拥有所有权限
            if "*" in user_permissions:
                return await func(*args, current_user=current_user, **kwargs)
            
            # 检查是否拥有所需权限
            for perm in required_permissions:
                if perm not in user_permissions:
                    raise HTTPException(
                        status_code=status.HTTP_403_FORBIDDEN,
                        detail=f"缺少权限: {perm}"
                    )
            
            return await func(*args, current_user=current_user, **kwargs)
        return wrapper
    return decorator

def require_roles(required_roles: List[Role]):
    """
    角色检查装饰器
    
    Usage:
        @router.get("/admin/dashboard")
        @require_roles([Role.ADMIN, Role.SUPER_ADMIN])
        async def admin_dashboard():
            ...
    """
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, current_user=Depends(get_current_user), **kwargs):
            if current_user.role not in required_roles:
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail="权限不足"
                )
            return await func(*args, current_user=current_user, **kwargs)
        return wrapper
    return decorator
```

### 4.3 资源所有权检查

```python
# core/security.py

async def check_resource_ownership(
    user_id: str,
    resource_user_id: str,
    user_role: Role
) -> bool:
    """
    检查用户是否有权访问资源
    
    规则:
    - 用户可以访问自己的资源
    - 管理员可以访问所有资源

    if user_role in [Role.ADMIN, Role.SUPER_ADMIN]:
        return True
    
    return user_id == resource_user_id
```

## 6. 安全机制

### 6.1 密码安全

```python
# core/security.py

import bcrypt

class PasswordManager:
    """密码管理器"""
    
    @staticmethod
    def hash_password(password: str) -> str:
        """加密密码"""
        salt = bcrypt.gensalt(rounds=12)
        hashed = bcrypt.hashpw(password.encode('utf-8'), salt)
        return hashed.decode('utf-8')
    
    @staticmethod
    def verify_password(plain_password: str, hashed_password: str) -> bool:
        """验证密码"""
        return bcrypt.checkpw(
            plain_password.encode('utf-8'),
            hashed_password.encode('utf-8')
        )
    
    @staticmethod
    def validate_password_strength(password: str) -> tuple[bool, str]:
        """
        验证密码强度
        
        规则:
        - 最少8位
        - 包含大小写字母
        - 包含数字
        - 可选: 包含特殊字符
        """
        if len(password) < 8:
            return False, "密码长度至少8位"
        
        if not any(c.isupper() for c in password):
            return False, "密码必须包含大写字母"
        
        if not any(c.islower() for c in password):
            return False, "密码必须包含小写字母"
        
        if not any(c.isdigit() for c in password):
            return False, "密码必须包含数字"
        
        return True, "密码强度合格"
```

### 6.2 Token黑名单机制

```python
# core/token_blacklist.py

import redis
from datetime import timedelta

class TokenBlacklist:
    """Token黑名单管理"""
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.prefix = "blacklist:token:"
    
    async def add_token(self, jti: str, expires_in: int):
        """
        将Token加入黑名单
        
        Args:
            jti: Token唯一标识
            expires_in: Token剩余有效时间(秒)
        """
        key = f"{self.prefix}{jti}"
        await self.redis.setex(key, expires_in, "revoked")
    
    async def is_blacklisted(self, jti: str) -> bool:
        """检查Token是否在黑名单"""
        key = f"{self.prefix}{jti}"
        return await self.redis.exists(key)
    
    async def revoke_user_tokens(self, user_id: str):
        """撤销用户所有Token"""
        # 从Redis获取用户所有活跃会话
        session_keys = await self.redis.keys(f"session:user:{user_id}:*")
        
        for key in session_keys:
            session_data = await self.redis.hgetall(key)
            if "jti" in session_data:
                await self.add_token(
                    session_data["jti"],
                    session_data.get("expires_in", 7200)
                )
            await self.redis.delete(key)
```

### 6.3 防暴力破解

```python
# core/rate_limiter.py

class LoginRateLimiter:
    """登录频率限制"""
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.max_attempts = 5  # 最大尝试次数
        self.lockout_duration = 900  # 锁定时长(秒) 15分钟
    
    async def check_and_increment(self, identifier: str) -> tuple[bool, int]:
        """
        检查并增加失败次数
        
        Args:
            identifier: 用户标识(username或IP)
        
        Returns:
            (是否允许登录, 剩余尝试次数)
        """
        key = f"login:attempts:{identifier}"
        attempts = await self.redis.get(key)
        
        if attempts is None:
            attempts = 0
        else:
            attempts = int(attempts)
        
        if attempts >= self.max_attempts:
            ttl = await self.redis.ttl(key)
            return False, 0
        
        return True, self.max_attempts - attempts
    
    async def record_failure(self, identifier: str):
        """记录登录失败"""
        key = f"login:attempts:{identifier}"
        await self.redis.incr(key)
        await self.redis.expire(key, self.lockout_duration)
    
    async def reset(self, identifier: str):
        """重置失败次数(登录成功时调用)"""
        key = f"login:attempts:{identifier}"
        await self.redis.delete(key)
```

### 6.4 会话管理

```python
# core/session_manager.py

class SessionManager:
    """会话管理器"""
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.max_sessions_per_user = 5
    
    async def create_session(
        self,
        user_id: str,
        access_token: str,
        refresh_token: str,
        device_type: str,
        expires_in: int
    ) -> str:
        """创建会话"""
        session_id = str(uuid.uuid4())
        key = f"session:user:{user_id}:{session_id}"
        
        session_data = {
            "access_token": access_token,
            "refresh_token": refresh_token,
            "device_type": device_type,
            "created_at": datetime.now().isoformat(),
        }
        
        await self.redis.hset(key, mapping=session_data)
        await self.redis.expire(key, expires_in)
        
        # 检查会话数量限制
        await self._enforce_session_limit(user_id)
        
        return session_id
    
    async def _enforce_session_limit(self, user_id: str):
        """强制执行会话数量限制"""
        pattern = f"session:user:{user_id}:*"
        sessions = await self.redis.keys(pattern)
        
        if len(sessions) > self.max_sessions_per_user:
            # 删除最旧的会话
            sessions_with_time = []
            for session_key in sessions:
                created_at = await self.redis.hget(session_key, "created_at")
                sessions_with_time.append((session_key, created_at))
            
            sessions_with_time.sort(key=lambda x: x[1])
            
            # 删除超出限制的旧会话
            for session_key, _ in sessions_with_time[:-self.max_sessions_per_user]:
                await self.redis.delete(session_key)
    
    async def get_user_sessions(self, user_id: str) -> list:
        """获取用户所有会话"""
        pattern = f"session:user:{user_id}:*"
        session_keys = await self.redis.keys(pattern)
        
        sessions = []
        for key in session_keys:
            session_data = await self.redis.hgetall(key)
            sessions.append(session_data)
        
        return sessions
    
    async def revoke_session(self, user_id: str, session_id: str):
        """撤销指定会话"""
        key = f"session:user:{user_id}:{session_id}"
        await self.redis.delete(key)
```

## 7. API认证中间件

```python
# core/dependencies.py

from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from jose import JWTError, jwt

security = HTTPBearer()

async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(security),
    db: Session = Depends(get_db),
    redis: Redis = Depends(get_redis)
) -> User:
    """
    获取当前登录用户
    
    验证流程:
    1. 提取Token
    2. 验证Token签名
    3. 检查Token是否过期
    4. 检查Token是否在黑名单
    5. 从数据库获取用户信息
    6. 检查用户状态
    """
    token = credentials.credentials
    
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="无效的认证凭证",
        headers={"WWW-Authenticate": "Bearer"},
    )
    
    try:
        # 解码Token
        payload = jwt.decode(
            token,
            JWT_CONFIG["SECRET_KEY"],
            algorithms=[JWT_CONFIG["ALGORITHM"]]
        )
        
        user_id: str = payload.get("sub")
        jti: str = payload.get("jti")
        token_type: str = payload.get("type")
        
        if user_id is None or token_type != "access":
            raise credentials_exception
        
        # 检查黑名单
        blacklist = TokenBlacklist(redis)
        if await blacklist.is_blacklisted(jti):
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Token已被撤销"
            )
        
    except JWTError:
        raise credentials_exception
    
    # 从数据库获取用户
    user = db.query(User).filter(
        User.id == user_id,
        User.deleted_at.is_(None)
    ).first()
    
    if user is None:
        raise credentials_exception
    
    # 检查用户状态
    if user.status != "active":
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="账号已被禁用"
        )
    
    return user

async def get_current_active_user(
    current_user: User = Depends(get_current_user)
) -> User:
    """获取当前活跃用户(额外检查)"""
    if current_user.status != "active":
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="用户账号未激活"
        )
    return current_user
```

```
POST   /api/v1/auth/password/reset    # 重置密码(忘记密码)

POST   /api/v1/auth/wechat/login      # 微信登录

# 用户信息

```

## 9. 安全最佳实践

### 9.1 生产环境配置
- 使用强随机密钥 (至少32字节)
- 启用HTTPS
- 配置CORS白名单
- 启用请求日志审计
- 定期轮换JWT密钥

### 9.2 Token安全
- 不在URL中传递Token
- 使用HttpOnly Cookie (可选方案)
- 实施Token刷新机制
- 敏感操作要求重新认证

### 9.3 监控告警
- 异常登录检测 (异地登录、频繁登录)
- 暴力破解告警
- Token异常使用告警
- 权限越权尝试告警
