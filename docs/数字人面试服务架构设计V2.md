# 数字人面试服务架构设计 V2

## 1. 概述

### 1.1 设计目标

本文档描述数字人面试服务的重新设计方案，核心目标：

- **共享数据模型**：与主服务共享同一套 SQLAlchemy 数据模型和向量数据库（ChromaDB）
- **独立运行**：作为独立的 FastAPI 服务运行，拥有独立的进程和端口
- **HTTP 通信**：服务间通过 HTTP RESTful API 进行通信
- **松耦合**：保持服务边界清晰，便于独立开发、部署和扩展

### 1.2 架构变更

| 方面 | 原设计 (V1) | 新设计 (V2) |
|------|------------|------------|
| 数据模型 | 各自独立定义 | **共享 `shared/models`** |
| 向量数据库 | 各自实例 | **共享同一 ChromaDB 实例** |
| 服务框架 | OpenAvatarChat | **独立 FastAPI 服务** |
| 通信方式 | HTTP + Redis Pub/Sub | **纯 HTTP RESTful** |
| 部署方式 | 单独容器 | 独立容器，共享数据卷 |

```python
from datetime import datetime
from typing import Optional
from uuid import uuid4

from sqlalchemy import DateTime, String
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column


def generate_uuid() -> str:
    return str(uuid4())


class Base(DeclarativeBase):
    """SQLAlchemy 声明式基类"""
    pass


class TimestampMixin:
    """时间戳 Mixin"""
    created_at: Mapped[datetime] = mapped_column(
        DateTime, 
        default=datetime.utcnow,
        nullable=False
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime,
        default=datetime.utcnow,
        onupdate=datetime.utcnow,
        nullable=False
    )


class SoftDeleteMixin:
    """软删除 Mixin"""
    deleted_at: Mapped[Optional[datetime]] = mapped_column(
        DateTime, 
        nullable=True
    )
```

```python
from shared.config.settings import get_settings

settings = get_settings()

# 创建共享引擎 - 两个服务使用相同配置

    echo=settings.DEBUG,
    pool_size=10,
    max_overflow=20
)

AsyncSessionLocal = async_sessionmaker(
    engine,
    class_=AsyncSession,
    expire_on_commit=False
)


async def get_db_session() -> AsyncSession:
    """获取数据库会话"""
    async with AsyncSessionLocal() as session:
        yield session
```

```python
import chromadb
from chromadb.config import Settings
from shared.config.settings import get_settings

settings = get_settings()

# ChromaDB 客户端 - 共享实例
_chroma_client = None


def get_chroma_client() -> chromadb.Client:
    """获取 ChromaDB 客户端单例"""
    global _chroma_client
    
    if _chroma_client is None:
        _chroma_client = chromadb.HttpClient(
            host=settings.CHROMA_HOST,
            port=settings.CHROMA_PORT,
            settings=Settings(
                anonymized_telemetry=False,
                allow_reset=False
            )
        )
    
    return _chroma_client


def get_knowledge_collection():
    """获取知识库集合"""
    client = get_chroma_client()
    return client.get_or_create_collection(
        name="knowledge_documents",
        metadata={"hnsw:space": "cosine"}
    )


def get_question_collection():
    """获取题目向量集合"""
    client = get_chroma_client()
    return client.get_or_create_collection(
        name="questions",
        metadata={"hnsw:space": "cosine"}
    )
```

### 5.2 核心服务类

#### 5.2.1 主服务 HTTP 客户端

```python
# interview_avatar_service/clients/main_service.py

import httpx
from typing import Optional
from shared.config.settings import get_settings

settings = get_settings()


class MainServiceClient:
    """主服务 HTTP 客户端"""
    
    def __init__(self):
        self.base_url = settings.MAIN_SERVICE_URL
        self.timeout = httpx.Timeout(30.0)
        self.headers = {
            "X-Service-Token": settings.INTERNAL_SERVICE_TOKEN,
            "Content-Type": "application/json"
        }
    
    async def start_interview_session(
        self,
        user_id: str,
        position_id: str,
        config: dict
    ) -> dict:
        """在主服务创建面试会话"""
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.post(
                f"{self.base_url}/api/v1/interviews/sessions/start",
                headers=self.headers,
                json={
                    "user_id": user_id,
                    "position_id": position_id,
                    "config": config
                }
            )
            response.raise_for_status()
            return response.json()["data"]
    
    async def fetch_next_question(
        self,
        session_id: str,
        context: dict
    ) -> Optional[dict]:
        """获取下一个问题"""
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.post(
                f"{self.base_url}/api/v1/questions/next",
                headers=self.headers,
                json={
                    "session_id": session_id,
                    "context": context
                }
            )
            if response.status_code == 404:
                return None  # 没有更多问题
            response.raise_for_status()
            return response.json()["data"]
    
    async def submit_answer(
        self,
        session_id: str,
        question_id: str,
        answer: dict,
        metadata: dict
    ) -> dict:
        """提交用户答案"""
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.post(
                f"{self.base_url}/api/v1/interviews/answers",
                headers=self.headers,
                json={
                    "session_id": session_id,
                    "question_id": question_id,
                    "answer": answer,
                    "metadata": metadata
                }
            )
            response.raise_for_status()
            return response.json()["data"]
    
    async def trigger_evaluation(
        self,
        session_id: str,
        qa_record_id: str,
        require_feedback: bool = True
    ) -> dict:
        """触发实时评估"""
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.post(
                f"{self.base_url}/api/v1/evaluations/realtime",
                headers=self.headers,
                json={
                    "session_id": session_id,
                    "qa_record_id": qa_record_id,
                    "require_feedback": require_feedback
                }
            )
            response.raise_for_status()
            return response.json()["data"]
    
    async def search_knowledge(
        self,
        query: str,
        position_id: str,
        top_k: int = 5
    ) -> list:
        """RAG 知识检索"""
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.post(
                f"{self.base_url}/api/v1/knowledge/search",
                headers=self.headers,
                json={
                    "query": query,
                    "position_id": position_id,
                    "top_k": top_k
                }
            )
            response.raise_for_status()
            return response.json()["data"]["results"]
    
    async def complete_interview(
        self,
        session_id: str,
        summary: dict
    ) -> dict:
        """完成面试"""
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.post(
                f"{self.base_url}/api/v1/interviews/sessions/{session_id}/complete",
```

```python
# interview_avatar_service/services/session_manager.py

from typing import Dict, Optional
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum

from shared.database.postgresql import AsyncSessionLocal
from shared.models.interview import InterviewSession


class SessionPhase(str, Enum):
    INITIALIZING = "initializing"
    GREETING = "greeting"
    QUESTIONING = "questioning"
    WAITING_ANSWER = "waiting_answer"
    PROCESSING = "processing"
    FEEDBACK = "feedback"
    ENDING = "ending"
    COMPLETED = "completed"


@dataclass
class ActiveSession:
    """活跃会话状态"""
    session_id: str
    user_id: str
    position_id: str
    phase: SessionPhase = SessionPhase.INITIALIZING
    current_question_id: Optional[str] = None
    current_question_order: int = 0
    answered_questions: list = field(default_factory=list)
    dialogue_history: list = field(default_factory=list)
    started_at: datetime = field(default_factory=datetime.utcnow)
    last_activity: datetime = field(default_factory=datetime.utcnow)


class SessionManager:
    """面试会话管理器"""
    
    def __init__(self):
        self._active_sessions: Dict[str, ActiveSession] = {}
    
    async def create_session(
        self,
        session_id: str,
        user_id: str,
        position_id: str
    ) -> ActiveSession:
        """创建新的活跃会话"""
        session = ActiveSession(
            session_id=session_id,
            user_id=user_id,
            position_id=position_id
        )
        self._active_sessions[session_id] = session
        return session
    
    def get_session(self, session_id: str) -> Optional[ActiveSession]:
        """获取活跃会话"""
        return self._active_sessions.get(session_id)
    
    def update_phase(self, session_id: str, phase: SessionPhase):
        """更新会话阶段"""
        if session := self._active_sessions.get(session_id):
            session.phase = phase
            session.last_activity = datetime.utcnow()
    
    def set_current_question(self, session_id: str, question_id: str):
        """设置当前问题"""
        if session := self._active_sessions.get(session_id):
            session.current_question_id = question_id
            session.current_question_order += 1
            session.last_activity = datetime.utcnow()
    
    def record_answer(self, session_id: str, question_id: str):
        """记录已回答的问题"""
        if session := self._active_sessions.get(session_id):
            session.answered_questions.append(question_id)
            session.current_question_id = None
            session.last_activity = datetime.utcnow()
    
    def add_dialogue(self, session_id: str, role: str, content: str):
        """添加对话记录"""
        if session := self._active_sessions.get(session_id):
            session.dialogue_history.append({
                "role": role,
                "content": content,
                "timestamp": datetime.utcnow().isoformat()
            })
            session.last_activity = datetime.utcnow()
    
    async def end_session(self, session_id: str) -> Optional[dict]:
        """结束会话"""
        if session := self._active_sessions.pop(session_id, None):
            return {
                "session_id": session.session_id,
                "questions_answered": len(session.answered_questions),
                "duration_seconds": (datetime.utcnow() - session.started_at).total_seconds()
            }
        return None
    
    def get_active_count(self) -> int:
        """获取活跃会话数量"""
        return len(self._active_sessions)
```

#### 5.2.3 对话引擎

```python
# interview_avatar_service/services/dialogue_engine.py

from typing import Optional
from shared.config.settings import get_settings
from .session_manager import session_manager, SessionPhase
from ..clients.main_service import MainServiceClient

settings = get_settings()


class DialogueEngine:
    """面试对话引擎"""
    
    def __init__(self):
        self.main_client = MainServiceClient()
    
    async def start_interview(self, session_id: str) -> str:
        """开始面试，返回欢迎语"""
        session = session_manager.get_session(session_id)
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        session_manager.update_phase(session_id, SessionPhase.GREETING)
        
        # 生成欢迎语
        greeting = await self._generate_greeting(session)
        session_manager.add_dialogue(session_id, "interviewer", greeting)
        
        return greeting
    
    async def present_next_question(self, session_id: str) -> Optional[dict]:
        """展示下一个问题"""
        session = session_manager.get_session(session_id)
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        # 从主服务获取下一题
        context = {
            "answered_questions": session.answered_questions,
            "current_order": session.current_question_order
        }
        
        question = await self.main_client.fetch_next_question(
            session.session_id,
            context
        )
        
        if not question:
            # 没有更多问题，结束面试
            session_manager.update_phase(session_id, SessionPhase.ENDING)
            return None
        
        # 更新会话状态
        session_manager.set_current_question(session_id, question["question_id"])
        session_manager.update_phase(session_id, SessionPhase.QUESTIONING)
        
        # 转换为自然语言
        natural_question = await self._naturalize_question(question)
        session_manager.add_dialogue(session_id, "interviewer", natural_question)
        
        return {
            "question_id": question["question_id"],
            "content": natural_question,
            "original": question
        }
    
    async def process_answer(
        self,
        session_id: str,
        answer_text: str,
        audio_url: Optional[str] = None
    ) -> dict:
        """处理用户答案"""
        session = session_manager.get_session(session_id)
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        session_manager.update_phase(session_id, SessionPhase.PROCESSING)
        session_manager.add_dialogue(session_id, "candidate", answer_text)
        
        # 提交答案到主服务
        answer_data = {
            "text": answer_text,
            "audio_url": audio_url
        }
        
        result = await self.main_client.submit_answer(
            session.session_id,
            session.current_question_id,
            answer_data,
            {"question_order": session.current_question_order}
        )
        
        # 触发评估
        evaluation = await self.main_client.trigger_evaluation(
            session.session_id,
            result["qa_record_id"]
        )
        
        # 记录已回答
        session_manager.record_answer(session_id, session.current_question_id)
        
        # 生成反馈
        session_manager.update_phase(session_id, SessionPhase.FEEDBACK)
        feedback = await self._generate_feedback(evaluation)
        session_manager.add_dialogue(session_id, "interviewer", feedback)
        
        return {
            "evaluation": evaluation,
            "feedback": feedback
        }
    
    async def end_interview(self, session_id: str) -> str:
        """结束面试，返回结束语"""
        session = session_manager.get_session(session_id)
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        session_manager.update_phase(session_id, SessionPhase.COMPLETED)
        
        # 通知主服务完成面试
        summary = await session_manager.end_session(session_id)
        await self.main_client.complete_interview(session_id, summary)
        
        # 生成结束语
        closing = await self._generate_closing()
        
        return closing
    
    async def _generate_greeting(self, session) -> str:
        """生成欢迎语"""
        return f"你好，欢迎参加本次面试。我是你的AI面试官，接下来我们将进行技术面试环节。准备好了就告诉我，我们开始吧。"
    
    async def _naturalize_question(self, question: dict) -> str:
        """将题目转换为自然口语表达"""
        # TODO: 使用 LLM 优化问题表达
        return question["content"]
    
    async def _generate_feedback(self, evaluation: dict) -> str:
        """根据评估生成简短反馈"""
        score = evaluation.get("score", 0)
        if score >= 80:
            return "很好，你的回答很全面。让我们继续下一个问题。"
        elif score >= 60:
            return "好的，我了解了。我们继续吧。"
        else:
            return "明白了。让我们进入下一个问题。"
    
    async def _generate_closing(self) -> str:
        """生成结束语"""
        return "感谢你参加本次面试，我们的面试到此结束。评估报告会在稍后生成，祝你一切顺利！"
```

```python
# shared/database/chromadb.py

from enum import Enum
from typing import List, Optional
import chromadb
from shared.config.settings import get_settings

settings = get_settings()


class ServiceRole(str, Enum):
    MAIN_SERVICE = "main"      # 可读写
    AVATAR_SERVICE = "avatar"  # 只读
    ADMIN = "admin"            # 完全控制


class ChromaDBManager:
    """ChromaDB 管理器"""
    
    def __init__(self, role: ServiceRole):
        self.role = role
        self._client = chromadb.HttpClient(
            host=settings.CHROMA_HOST,
            port=settings.CHROMA_PORT
        )
    
    def _check_write_permission(self):
        """检查写入权限"""
        if self.role == ServiceRole.AVATAR_SERVICE:
            raise PermissionError("Avatar service has read-only access to ChromaDB")
    
    def get_knowledge_collection(self):
        """获取知识库集合"""
        return self._client.get_or_create_collection(
            name="knowledge_documents",
            metadata={"hnsw:space": "cosine"}
        )
    
    def search_knowledge(
        self,
        query_embedding: List[float],
        n_results: int = 5,
        where: Optional[dict] = None
    ) -> dict:
        """搜索知识库"""
        collection = self.get_knowledge_collection()
        return collection.query(
            query_embeddings=[query_embedding],
            n_results=n_results,
            where=where,
            include=["documents", "metadatas", "distances"]
        )
    
    def add_knowledge(
        self,
        ids: List[str],
        embeddings: List[List[float]],
        documents: List[str],
        metadatas: List[dict]
    ):
        """添加知识文档 (仅主服务可用)"""
        self._check_write_permission()
        collection = self.get_knowledge_collection()
        collection.add(
            ids=ids,
            embeddings=embeddings,
            documents=documents,
            metadatas=metadatas
        )
    
    def search_questions(
        self,
        query_embedding: List[float],
        n_results: int = 10,
        where: Optional[dict] = None
    ) -> dict:
        """搜索相似题目"""
        collection = self._client.get_or_create_collection(
            name="questions",
            metadata={"hnsw:space": "cosine"}
        )
        return collection.query(
            query_embeddings=[query_embedding],
            n_results=n_results,
            where=where,
            include=["documents", "metadatas", "distances"]
        )
```

```python
# shared/config/logging.py

import logging
import sys
from pythonjsonlogger import jsonlogger


def setup_logging(service_name: str):
    """配置结构化日志"""
    logger = logging.getLogger()
    logger.setLevel(logging.INFO)
    
    handler = logging.StreamHandler(sys.stdout)
    formatter = jsonlogger.JsonFormatter(
        fmt="%(asctime)s %(levelname)s %(name)s %(message)s",
        rename_fields={"asctime": "timestamp", "levelname": "level"}
    )
    formatter.add_fields = lambda log_record, _, __: log_record.update({
        "service": service_name
    })
    handler.setFormatter(formatter)
    logger.addHandler(handler)
```
